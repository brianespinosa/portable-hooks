  state = { tags: [], newLabel: '' };

    this.setState({
      tags: [...this.state.tags, name],

    });
  }

  removeTagAtIndex(index) {

      ...this.state,
      tags: omitIndexes([index], this.state.tags),

      tags: move(index, index - 1, this.state.tags),
    });

      ...this.state,

  }

  }
}
